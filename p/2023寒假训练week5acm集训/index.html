			if(num[i]>j){//Á¨¨‰∏Ä‰∏™Ë¶ÅÊ∞¥Êô∂Êï∞ÈáèÂ§ß 
				dp[i][j]=dp[i-1][j]; 
			}
			else{
				dp[i][j]=max(dp[i-1][j],dp[i-1][j-num[i]]+val[i]);
			}
		}
	}
	cout<<dp[m][n]<<endl;
	return 0;
